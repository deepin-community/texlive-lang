\NewDocumentCommand\TrioCalculs{mmm}{%
  Les combinaisons Trio de #1, #2 et #3 sont :%
  \begin{enumerate}%
  \item $#1\times#2+#3=\fpeval{#1*#2+#3}$%
  \item $#1\times#2-#3=\fpeval{#1*#2-#3}$%
  \item $#2\times#3+#1=\fpeval{#2*#3+#1}$%
  \item $#2\times#3-#1=\fpeval{#2*#3-#1}$%
  \item $#3\times#1+#2=\fpeval{#3*#1+#2}$%
  \item $#3\times#1-#2=\fpeval{#3*#1-#2}$%
  \end{enumerate}%
}%
